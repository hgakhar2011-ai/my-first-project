<script>
const qa = [
    {
        question: "What is PPE?",
        answer: "PPE stands for Personal Protective Equipment, including gloves, masks, and gowns."
    },
    {
        question: "How do you assess LoC?",
        answer: "Using the AVPU scale: Alert, Voice, Pain, Unresponsive."
    },
    {
        question: "What is the purpose of the Ben Stelter Foundation?",
        answer: "To support children fighting cancer and continue Ben's legacy."
    }
];

let current = null;
let showingAnswer = false;

function newCard() {
    current = qa[Math.floor(Math.random() * qa.length)];
    document.getElementById("card").innerText = current.question;
    showingAnswer = false;
}

document.getElementById("card").addEventListener("click", () => {
    if (!current) return;
    if (!showingAnswer) {
        document.getElementById("card").innerText = current.answer;
        showingAnswer = true;
    } else {
        document.getElementById("card").innerText = current.question;
        showingAnswer = false;
    }
});
</script>
